00098     //A barycentric coordinate of a spline is define by its curvilinear coordinate and its radius
00099     //in baryP argument here, the first is curvilinear, second is radius, third is not use for instance
00100     int addBaryPoint(const Vector3& baryP, int splineId, Real& r)
00101     {
00102         int i = nbp++;
00103         if ((int)outmodel->getX()->size() <= i)
00104             outmodel->resize(i+1);
00105         if (mapping)
00106         {
00107                 CubicBezierCurve<1> cb(this->model, splineId);
00108                 int localsplineId;
00109                 int edgeId = model->comuteEdgeIdFromSplineId(splineId,localsplineId);
00110                 if ( cb.isCubicBezier() )
00111                         i = mapping->addBaryPoint(edgeId,baryP,true);
00112                 else
00113                         i = mapping->addBaryPoint(edgeId,baryP,false);
00114             r=cb.r();
00115         }
00116         else
00117         {
00118             std::cout << "WARNING[AdaptiveBeamContactMapper] AdaptiveBeamMapping is not created" << std::endl;
00119         }
00120         return i;
00121     }
